    <div id="slideshow">
        <img id="portfolio-image" src="portfolio-sprite.png" alt="Portfolio">
    </div>
    <div class="navigation">
        <button onclick="prevImage()">Previous</button>
        <button onclick="nextImage()">Next</button>
    </div>

    <script>
        // 이미지 슬라이드쇼 로직
        const totalPages = 8;  // 포트폴리오 페이지 수
        const imageHeight = 600; // 각 페이지의 높이 (CSS와 동일하게 유지)
        let currentIndex = 0;

        function updateImage() {
            const image = document.getElementById('portfolio-image');
            image.style.transform = `translateY(${-currentIndex * imageHeight}px)`;
        }

        function prevImage() {
            currentIndex = (currentIndex > 0) ? currentIndex - 1 : totalPages - 1;
            updateImage();
        }

        function nextImage() {
            currentIndex = (currentIndex < totalPages - 1) ? currentIndex + 1 : 0;
            updateImage();
        }
    </script>
